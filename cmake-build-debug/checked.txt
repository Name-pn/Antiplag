Текст программы:
program lab5;
  const
     N = 3;
  type
     t_matrix = array [1..N,1..N] of longint;

  {Процедура ввода матрицы}
  procedure input_matrix(var a: t_matrix);
    var
       i,j: longint;
    begin
      for j:=1 to N do
         begin
           for i:=1 to N do
              begin
                 read(a[i,j]);
                 write(' ');
              end;
         writeln;
         end;
    end;

  {Процедура для получения суммы двух матриц }
  procedure sum(const a, b: t_matrix; var c: t_matrix);
    var
      i, j:longint;
    begin
       for i:=1 to N do
          for j:=1 to N do
             c[i,j]:=a[i,j]+b[i,j];
    end;

  {Функция возвращает произведение столбца матрицы на массив}
  function multi_el ( a, b: t_matrix; c, r: longint):longint;
    var
       k, sum: longint;
    begin
       sum := 0;
       for k := 1 to N do
          sum := a[r, k] * b[k, c] + sum;
       multi_el := sum;
    end;

  {Процедура находит произведение матриц}
  procedure multiplication( a, b: t_matrix; var c: t_matrix);
     var
        i, j: longint;
     begin
        for i := 1 to N do
           for j := 1 to  N do
              c[i, j] := multi_el(a, b, j, i);
     end;

  {Процедура выводит матрицу}
  procedure output_matrix (const a: t_matrix);
    var
     i, j: longint;
    begin
       for i := 1 to N do
       begin
          for j := 1 to N do
             begin
                write(a[i,j]);
                write(' ');
             end;
       writeln;
       writeln;
       end;
    end;

  var
     a,b,c: t_matrix;
     i,p: longint;
   begin
     writeln('Введите n');
     read( p );
     writeln('Введите A');
     input_matrix(a);
     b := a;
     c := a;
     for i:= 2 to p do
        begin
           multiplication( a, b, b );
           sum( b, c, c );
        end;
     writeln('Получена матрица B')
     output_matrix( c );
   end.
